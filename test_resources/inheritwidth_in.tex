Test whether a subfigures pass their width on to inner graphics.

\begin{subfigure}[bt]{0.5\linewidth}
    % If included graphics has a width not relative to containing box, it
    % should not change width
    \includegraphics{file.png}
    % Should change if inherited from parent
    \includegraphics[width=\linewidth]{file2.pdf}
    % Should change if relative to parent
    \includegraphics[width=0.6\linewidth]{file2.pdf}
    % Content should change even if commented out
%    \includegraphics[width=0.8\linewidth]{file2-alt.pdf}
    % Other content should change too
    \otherthing[width=\linewidth] {random stuff}
\end{subfigure}

% Values should not change if subfigure is 100% relative to something
\begin{subfigure}[bt]{\linewidth}
    % Should change if relative to parent
    \includegraphics[width=0.6\linewidth]{file2.pdf}
\end{subfigure}

% subfigure can have any width setting, not just \linewidth
\begin{subfigure}[bt]{0.5\pagewidth}
    % Should change if relative to parent
    \includegraphics[width=0.6\linewidth]{file2.pdf}
    % Other content should change too
    \otherthing[width=\linewidth] {random stuff}
\end{subfigure}
